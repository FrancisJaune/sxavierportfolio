#function to turn estimated players into a number
#splits the a string range in the form  "x - y" into two integers x and y, then returns the average of the two integers
def players(estimated):
    a = estimated.split(" - ")

    b1 = int(a[0])
    b2 = int(a[1])

    c = (b1 + b2) / 2
    return c
